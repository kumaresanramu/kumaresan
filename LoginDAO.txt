public class LoginDAO {
Connection cn=null;
PreparedStatement ps=null;
String sql="insert into LOGIN_MS values(?,?)";
String sql1="delete from LOGIN_MS where username=?";
String sql2="update LOGin_ms set password=? where username=?";
public void insertData(LoginDTO ldto)
{
    try
    {
        ConnectionFactory con=new ConnectionFactory();
        cn=con.getConn();
        ps=cn.prepareStatement(sql);
        ps.setString(1,ldto.getUname());
        ps.setString(2,ldto.getPass());
        ps.executeUpdate();
        cn.commit();
    }
     
    catch(SQLException se)
    {
        se.printStackTrace();
    }
}
public void deleteData(LoginDTO ldto)
{
    try
    {
        ConnectionFactory con=new ConnectionFactory();
        cn=con.getConn();
        ps=cn.prepareStatement(sql1);
        ps.setString(1,ldto.getUname());
        ps.executeUpdate();
        cn.commit();
    }
     
    catch(SQLException se)
    {
        se.printStackTrace();
    }
}
     
    public void updateData(LoginDTO ldto)
    {
        try
        {
            ConnectionFactory con=new ConnectionFactory();
            cn=con.getConn();
            ps=cn.prepareStatement(sql2);
            ps.setString(1,ldto.getPass());
            ps.setString(2,ldto.getUname());
            ps.executeUpdate();
            cn.commit();
        }
         
        catch(SQLException se)
        {
            se.printStackTrace();
        }
    }
}
